Contador crescente de 0 ate 10

Com uma função unmamed ou seja a função é escrita diretamente dentro de outra função


Contador: 0

                    button onclick="funcContador1()"> Crescente </button>

                    function funcContador1()
                    {
                        var numero = 0;
                        var t = setInterval(function()
                        {
                            document.getElementById("numeroDoContador1").innerHTML = numero;
                            if (numero === 10) clearInterval(t);  // se o numero for igual a 10 para o contador de tempo
                            numero++; // aumenta o numero em 1
                        },1000);
                    }

Contador decrescente de 10 ate 0

Com uma função named ou seja um função que chama outra função


Contador: 0

                    var n = 10;
                    var t;
                    function funcContador2()
                    {
                        t = setInterval(funcDecrescente, 1000);
                    }

                    function funcDecrescente()
                    {
                        document.getElementById("numeroDoContador2").innerHTML = n;
                        n--;
                        if (n === 0)
                            clearInterval(t);
                    }
